Variáveis de Sessão:

$_SESSION['PISO']= Array contendo os ID'd dos pisos selecionados separados por virgula
                    
 $_SESSION['PISO_CUSTOMIZADO'] = PIso adicionado na sessao do usuário

$_SESSION['NOME']= Array contendo os Nomes dos pisos selecionados separados por virgula
                    

$_SESSION['AREA']= Guarda a Area calculada

$_SESSION['PERIMETRO']= Guarda o Perimetro calculado

$_SESSION['PRODUTO']= Array contendo os ID'd dos produtos selecionados separados por virgula
                    

$_SESSION['ARGAMASSA']= Array contendo os ID'd das argamassas selecionados separados por virgula
                    

$_SESSION['REJUNTE']= Array contendo os ID'd dos rejuntes selecionados separados por virgula
